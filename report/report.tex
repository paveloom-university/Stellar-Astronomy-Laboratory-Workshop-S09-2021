–ù–∞—á–∞–ª—å–Ω—ã–µ –ø–æ–ª–æ–∂–µ–Ω–∏—è $ (R, \psi, Z) $ –≤ –ì–∞–ª–∞–∫—Ç–æ—Ü–µ–Ω—Ç—Ä–∏—á–µ—Å–∫–æ–π —Ü–∏–ª–∏–Ω–¥—Ä–∏—á–µ—Å–∫–æ–π —Å–∏—Å—Ç–µ–º–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –∏ –∏—Ö –ø—Ä–æ–∏–∑–≤–æ–¥–Ω—ã–µ –ø–æ –≤—Ä–µ–º–µ–Ω–∏ $ (\dot{R}, \dot{\psi}, \dot{Z}) $ –º–æ–≥—É—Ç –±—ã—Ç—å –æ–ø—Ä–µ–¥–µ–ª–µ–Ω—ã –ø–æ —Å–ª–µ–¥—É—é—â–∏–º —Ñ–æ—Ä–º—É–ª–∞–º:

\su
\begin{equation}
\begin{gathered}
  R = \sqrt{X^2 + Y^2}, \\
  \psi = \atantwo(Y, X), \\
  \dot{R} = -U \cos{\phi} + V \sin{\psi}, \\
  \dot{\psi} = (U \sin{\psi} + V \cos{\psi}) / R, \\
  \dot{Z} = W.
\end{gathered}
\end{equation}

–û—Ç—Å—é–¥–∞ –º—ã –æ–ø—Ä–µ–¥–µ–ª—è–µ–º –Ω–∞—á–∞–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è –∫–∞–Ω–æ–Ω–∏—á–µ—Å–∫–∏—Ö –º–æ–º–µ–Ω—Ç–æ–≤:

\su
\begin{equation}
\begin{aligned}
  p_R^0 = \dot{R}, \; p_\psi^0 = R^2 \dot{\psi}, \; p_Z^0 = \dot{Z}.
\end{aligned}
\end{equation}

–ù–∞–∫–æ–Ω–µ—Ü, –æ—Ä–±–∏—Ç–∞ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è –ø—É—Ç–µ–º –∏–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏—è —É—Ä–∞–≤–Ω–µ–Ω–∏–π –õ–∞–≥—Ä–∞–Ω–∂–∞:

\begin{equation}
\begin{gathered}
  \dot{R} = p_R, \\
  \dot{\psi} = p_\psi / R^2, \\
  \dot{Z} = p_Z, \\
  \dot{p_R} = - \partial \Phi(R, Z) / \partial R + p_\psi^2 / R^3, \\
  \dot{p_\psi} = 0, \\
  \dot{p_Z} = - \partial \Phi(R, Z) / \partial Z.
\end{gathered}
\end{equation}

–î–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –ø–æ–ª–Ω–æ–π —ç–Ω–µ—Ä–≥–∏–∏ –º—ã —Å–Ω–∞—á–∞–ª–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ–º —Ä–∞–¥–∏–∞–ª—å–Ω—É—é —Å–∫–æ—Ä–æ—Å—Ç—å

\su
\begin{equation}
\begin{aligned}
  \Pi = -U \frac{X}{R} + V \frac{Y}{R} = & -(-p_r \cos{\psi} + (p_\psi / R) \sin{\psi}) \cos{\psi} \; + \\
  & + (p_R \sin{\psi} - (p_\psi / R) \cos{\psi}) \sin{\psi},
\end{aligned}
\end{equation}

—Ç–∞–Ω–≥–µ–Ω—Ü–∏–∞–ª—å–Ω—É—é —Å–∫–æ—Ä–æ—Å—Ç—å

\su
\begin{equation}
\begin{aligned}

  & + (p_R \sin{\psi} - (p_\psi / R) \cos{\psi}) \cos{\psi},
\end{aligned}
\end{equation}

–∞ –∑–∞—Ç–µ–º –ø–æ–ª–Ω—É—é —Å–∫–æ—Ä–æ—Å—Ç—å $ V_\text{–ø–æ–ª–Ω.} = \sqrt{\Pi^2 + \Omega^2 + W^2} = \sqrt{\Pi^2 + \Omega^2 + p_Z^2} $. –ü–æ—Å–ª–µ —ç—Ç–æ–≥–æ –ø–æ–ª–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è –º–æ–∂–µ—Ç –±—ã—Ç—å –≤—ã—á–∏—Å–ª–µ–Ω–∞ –∫–∞–∫ $ E = \Psi(R, Z) + V_\text{–ø–æ–ª–Ω.}^2 / 2 $.
